package test;
import java.util.*;
import java.io.*;

/**
 *
 * @author HP
 */
public class Flying {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner reader = new Scanner(System.in); 
        System.out.println("Enter file name:");
        String fileName = reader.nextLine();
        
        try {
            
            String filePath = "c:/users/hp/desktop/"+fileName;
            File file = new File(filePath);
            
            if(file.length() == 0) {
                System.out.println("File is Empty, Program is Closing");
                System.exit(0);
            }
            else {
                // Read Manufacturer from user input
                System.out.println("Enter manufacturer:");
                String inputManufacturer = reader.nextLine().toLowerCase();
                
                // Read Model from user input
                System.out.println("Enter Model:");
                String inputModel = reader.nextLine().toLowerCase();
                
                Scanner scanner = new Scanner(file);
                
                while (scanner.hasNextLine()) {  

                    String s = scanner.nextLine();
                    
                    // Array String
                    String[] parts = s.split("\\^");

                    //Passengers in the plane
                    String passengers = parts[0];
                    //String to int
                    int noOfPassengers = Integer.parseInt(passengers);

                    // Range of the plane
                    String range = parts[1];
                    // String to int
                    int rangeInNumber = Integer.parseInt(range);

                    String name = parts[2];
                    String manufacturer = parts[3].toLowerCase();
                    String model = parts[4].toLowerCase();
                        
                    if(manufacturer.equals(inputManufacturer) & model.equals(inputModel)){
                        System.out.println("Name :"+name+" Model: "+model+" Manufacturer: "+manufacturer+" Range: "+rangeInNumber+"kms"+" Passengers: "+noOfPassengers);
                            
                    }
                    else {
                        System.out.println("No plane with the model "+inputModel+" made by the manufacturer "+inputManufacturer+" was found");
                    }

                }
                scanner.close();
            }
            
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }         
    }
    
    
    
}
